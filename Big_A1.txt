Write a program with Arduino UNO board to calculate the distance of a obstacle based on the Ultrasonic sensor inputs. If the distance calculated is less than a certain value turn on a buzzer / beeper with a LED in ON state and display the distance in LCD / OLED

#include <SPI.h>

const int trig=5;
const int echo=6;
long int duration;
int treshold=10;
int led=3;
int buz=4;
int distance;

#define OLED_MOSI   9
#define OLED_CLK   10
#define OLED_DC    11
#define OLED_CS    12
#define OLED_RESET 13
Adafruit_SSD1306 display(OLED_MOSI, OLED_CLK, OLED_DC, OLED_RESET, OLED_CS);

void setup()
{
  Serial.begin(9600);
  pinMode(trig,OUTPUT);
  pinMode(led,OUTPUT);
  pinMode(buz,OUTPUT);
  pinMode(echo,INPUT);
  
  display.begin(SSD1306_SWITCHCAPVCC);
  display.display();
  delay(2000);
  display.clearDisplay();
}

void loop()
{
display.clearDisplay();
  
  //ultrasonic sensor
  digitalWrite(trig,LOW);
  delay(50);
  
  digitalWrite(trig,HIGH);
  delay(50);
  digitalWrite(trig,LOW);

  duration=pulseIn(echo,HIGH);

  distance=duration*0.034/2;

  display.print("distance");
  display.println(distance);

//buzzer and led
  if(distance>10)
  {
    digitalWrite(buz,HIGH);
    digitalWrite(led,HIGH);
  }
  else
  {
    digitalWrite(buz,LOW);
    digitalWrite(led,LOW);
  }

//oled
  display.setTextSize(2);
  display.setTextColor(WHITE);
  display.setCursor(10,0);
  display.clearDisplay();
  display.println("distance");
  display.println(distance);
  display.display();

  
}
