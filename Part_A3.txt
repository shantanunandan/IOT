#include<DHT.h>
#include<Wire.h>
#include<SPI.h>

int led = 5;
int data = 3;

DHT dht(data, DHT11);

void setup() { 
  Serial.begin(9600);
  pinMode(led, OUTPUT);
  dht.begin();
}

void loop() {
	while(Serial.available() > 0) {
		String ch = Serial.readString();
		if(ch == "on") {
			digitalWrite(led, HIGH);
		}
		else if(ch == "off") {
			digitalWrite(led, LOW);
		}
		else if(ch == "temperature") {
			float t = dht.readTemperature();
			float h = dht.readHumidity();
			if (isnan(t) || isnan(h)) {
				Serial.print("Sensor Failed");
			}
			else {
				Serial.print("Humidity : ");
				Serial.print(h);
				Serial.print("Temprature : ");
				Serial.print(t);
			}
		}
	}
}

import serial

ser = serial.Serial("/dev/ttyAMC0", 9600)
while 1:
	print("1. LED On \n2. LED Off \n3. Temprature")
	ch = int(input("Enter Choice : "))
	if(ch == 1):
		ser.flush()
		ser.write(str.encode("on"))
	elif(ch == 2):
		ser.flush()
		ser.write(str.encode("off"))
	elif(ch == 3):
		ser.flush()
		ser.write(str.encode("temperature"))
		readData = ser.readLine()
		print(readData)
		readData = ser.readLine()
		print(readData)		
	else:
		break
